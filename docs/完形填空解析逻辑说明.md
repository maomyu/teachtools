### 1.3 单个试卷导入 import_paper() (第66-246行) - 核心函数

```python
async def import_paper(file_path: Path, batch_id: str, use_llm: bool = True) -> Dict:
    result = {
        "filename": file_path.name,
        "status": "pending",
        "error": None,
        "passages": 0,
        "cloze_points": 0
    }

    try:
        # =====================================================
        # Step 1: 解析文件名获取元数据
        # =====================================================
        parser = DocxParser(str(file_path))
        metadata = parser.parse_filename()

        # 从父目录提取区县信息（如果文件名中没有）
        parent_dir = file_path.parent
        if parent_dir.name in ["东城", "西城", "海淀", "朝阳", ...]:
            metadata["region"] = parent_dir.name

        async with async_session() as session:
            # 创建试卷记录
            paper = await get_or_create_paper(session, file_path.name, file_path, metadata)

            # =====================================================
            # Step 2: AI 解析文档（核心）
            # =====================================================
            llm_parser = LLMDocumentParser()
            llm_result = await llm_parser.parse_document(str(file_path))

            if not llm_result.success:
                paper.import_status = "failed"
                return result

            paper.import_status = "completed"
            paper.parse_strategy = "llm"
            paper.confidence = 0.95

            # 初始化话题分类器
            topic_classifier = TopicClassifier()
            grade = metadata.get("grade", "初二")

            # =====================================================
            # Step 3: 处理完形填空（重点）
            # =====================================================
            if llm_result.cloze and llm_result.cloze.get("found"):
                cloze_data = llm_result.cloze
                content_with_blanks = cloze_data.get("content_with_blanks", "")
                blanks = cloze_data.get("blanks", [])

                if content_with_blanks and blanks:
                    # 3.1 创建完形文章记录
                    cloze_passage = ClozePassage(
                        paper_id=paper.id,
                        content=content_with_blanks,
                        word_count=len(content_with_blanks.split())
                    )
                    session.add(cloze_passage)
                    await session.flush()

                    # 3.2 调用 AI 分类主题
                    topic_result = await topic_classifier.classify(content_with_blanks, grade=grade)
                    if topic_result.success:
                        cloze_passage.primary_topic = topic_result.primary_topic
                        cloze_passage.secondary_topics = json.dumps(topic_result.secondary_topics)
                        cloze_passage.topic_confidence = topic_result.confidence

                    # 3.3 初始化考点分析器
                    cloze_analyzer = ClozeAnalyzer()

                    # 3.4 遍历每个空格，分析考点
                    for blank in blanks:
                        blank_number = blank.get("blank_number")
                        options = blank.get("options", {})
                        correct_answer = blank.get("correct_answer")
                        correct_word = blank.get("correct_word")

                        # 调用 AI 分析考点类型
                        analysis = await cloze_analyzer.analyze_point(
                            blank_number=blank_number,
                            correct_word=correct_word or "",
                            options=options,
                            context=content_with_blanks
                        )

                        # 提取上下文句子
                        sentence = cloze_analyzer.extract_context(content_with_blanks, blank_number)

                        # 创建考点记录
                        cloze_point = ClozePoint(
                            cloze_id=cloze_passage.id,
                            blank_number=blank_number,
                            correct_answer=correct_answer,
                            correct_word=correct_word,
                            options=json.dumps(options, ensure_ascii=False),
                            point_type=analysis.point_type if analysis.success else None,
                            translation=analysis.translation,
                            explanation=analysis.explanation,
                            confusion_words=json.dumps(analysis.confusion_words),
                            sentence=sentence
                        )
                        session.add(cloze_point)
                        result["cloze_points"] += 1

            await session.commit()

    except Exception as e:
        result["status"] = "failed"
        result["error"] = str(e)

    return result
```

```python
async def upload_file(self, file_path: str) -> str:
    """上传文件到 DashScope，返回 fileid"""
    async with httpx.AsyncClient(timeout=120.0) as client:
        with open(file_path, 'rb') as f:
            files = {
                'file': (Path(file_path).name, f, 'application/vnd.openxmlformats-officedocument.wordprocessingml.document')
            }
            data = {'purpose': 'file-extract'}  # 必须指定
            headers = {'Authorization': f'Bearer {self.api_key}'}

            response = await client.post(self.UPLOAD_URL, headers=headers, files=files, data=data)

        if response.status_code != 200:
            raise Exception(f"上传文件失败: {response.text}")

        result = response.json()
        return result['id']  # 返回 fileid
```

```python
async def parse_document(self, file_path: str, use_fileid: bool = True) -> LLMParseResult:
    try:
        if use_fileid:
            # 方式1: 上传文件后使用 fileid 协议
            fileid = await self.upload_file(file_path)
            messages = [
                {"role": "system", "content": "You are a helpful assistant..."},
                {"role": "system", "content": f"fileid://{fileid}"},  # 关键：fileid协议
                {"role": "user", "content": self.EXTRACT_PROMPT}
            ]
        else:
            # 方式2: 直接读取文本（备用）
            from docx import Document
            doc = Document(file_path)
            full_text = "\n".join([p.text for p in doc.paragraphs])
            messages = [
                {"role": "system", "content": "You are a helpful assistant..."},
                {"role": "user", "content": f"{self.EXTRACT_PROMPT}\n\n{full_text[:30000]}"}
            ]

        # 调用 LLM API
        async with httpx.AsyncClient(timeout=120.0) as client:
            payload = {
                "model": "qwen-long",
                "messages": messages,
                "temperature": 0.1,    # 低温度保证稳定性
                "max_tokens": 16000    # 支持完整输出
            }

            response = await client.post(self.API_URL, headers=headers, json=payload)
            result = response.json()
            content = result['choices'][0]['message']['content']

            return self._parse_llm_response(content)

    except Exception as e:
        return LLMParseResult(success=False, error=str(e))
```

def _parse_llm_response(self, content: str) -> LLMParseResult:
    try:
        json_str = content

        # 提取 JSON（处理 markdown 代码块）
        if '```json' in content:
            start = content.find('```json') + 7
            end = content.find('```', start)
            json_str = content[start:end].strip()
        elif '```' in content:
            start = content.find('```') + 3
            end = content.find('```', start)
            json_str = content[start:end].strip()

        data = json.loads(json_str)

        return LLMParseResult(
            success=True,
            metadata=data.get('metadata', {}),
            passages=data.get('passages', []),
            cloze=data.get('cloze', {}),    # 完形填空数据
            raw_response=content
        )

    except json.JSONDecodeError as e:
        return LLMParseResult(success=False, error=f"JSON解析失败: {str(e)}")

### 3.5 上下文提取方法 extract_context() (第265-305行)

```python
def extract_context(self, content: str, blank_number: int, window: int = 100) -> str:
    """从文章中提取包含指定空格的上下文"""

    # 匹配不同格式的空格标记
    import re
    patterns = [
        rf'[{blank_number}]',       # ①②③格式
        rf'\({blank_number}\)',     # (1)(2)(3)格式
        rf'\[{blank_number}\]',     # [1][2][3]格式
    ]

    for pattern in patterns:
        match = re.search(pattern, content)
        if match:
            pos = match.start()
            # 提取前后文
            start = max(0, pos - window)
            end = min(len(content), pos + window)
            context = content[start:end]

            # 尝试提取完整句子
            sentences = re.split(r'[.!?]', context)
            if len(sentences) >= 2:
                for i, sent in enumerate(sentences):
                    if re.search(pattern, sent):
                        return sent.strip()

            return context.strip()

    # 找不到空格标记，返回文章开头
    return content[:window * 2]
```

```python
class ClozePassage(Base):
    __tablename__ = "cloze_passages"
```

```python
class ClozePoint(Base):
    __tablename__ = "cloze_points"

    id = Column(Integer, primary_key=True)
    cloze_id = Column(Integer, ForeignKey("cloze_passages.id"))

    # 空格基本信息
    blank_number = Column(Integer)          # 第几个空
    correct_answer = Column(String(100))    # A/B/C/D
    correct_word = Column(String(255))      # 正确答案词
    options = Column(Text)                  # JSON: {"A": "...", ...}

    # 考点分析结果（来自 ClozeAnalyzer）
    point_type = Column(String(50))         # 固定搭配/词义辨析/熟词僻义
    point_detail = Column(Text)             # 考点详解
    translation = Column(Text)              # 翻译
    explanation = Column(Text)              # 解析

    # 易混淆信息
    confusion_words = Column(Text)          # JSON格式

    # 上下文
    sentence = Column(Text)                 # 包含该空的句子
    char_position = Column(Integer)

    created_at = Column(DateTime)

    # 约束：考点类型必须是三类之一
    __table_args__ = (
        CheckConstraint(
            "point_type IN ('固定搭配', '词义辨析', '熟词僻义')",
            name="ck_point_type"
        ),
    )

    # 关系
    cloze = relationship("ClozePassage", back_populates="points")
```

```typescript
// 考点出现位置
export interface PointOccurrence {
  sentence: string          // 例句
  source: string            // 出处
  blank_number: number      // 空格序号
  point_type: string        // 考点类型
  explanation?: string      // 解析
  passage_id?: number       // 完形文章ID
}

// 考点汇总
export interface PointSummary {
  word: string              // 考点词
  definition?: string       // 释义
  frequency: number         // 出现次数
  point_type: string        // 考点类型
  occurrences: PointOccurrence[]  // 出现位置列表
}

// 考点汇总列表响应
export interface PointListResponse {
  total: number
  items: PointSummary[]
}
```
